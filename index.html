sendBtn.addEventListener('click', async () => { statusDiv.innerText = "Mengekstrak daftar item..."; sendBtn.disabled = true; // 1. Ekstraksi Tanggal Struk const matchTanggal = extractedText.match(/(\d{4}-\d{2}-\d{2})/); const tanggalStruk = matchTanggal ? matchTanggal[1] : "Tidak ditemukan"; // 2. Ekstraksi Multi-Item berdasarkan baris teks const lines = extractedText.split('\n'); const detectedItems = []; for (let i = 0; i < lines.length; i++) { const line = lines[i].trim(); // Regex untuk mencari pola perkalian (misal: "1 x 40,000" atau "1x 8.000") const matchQtyPrice = line.match(/^(\d+)\s*x\s*([\d\.,]+)/i); if (matchQtyPrice) { const qty = parseInt(matchQtyPrice[1]); // Bersihkan tanda titik/koma agar harga menjadi angka murni bulat const price = parseInt(matchQtyPrice[2].replace(/[\.,]/g, '')); // Mengambil nama produk dari baris tepat di atasnya let itemName = "Item Tanpa Nama"; if (i > 0) { let prevLine = lines[i - 1].trim(); // Pastikan baris di atasnya bukan baris kosong atau pembatas garis putus-putus if (prevLine && !prevLine.includes('---')) { itemName = prevLine; } } // Masukkan ke dalam daftar item yang berhasil dideteksi detectedItems.push({ name: itemName, qty: qty, price: price }); } } // Menyusun payload data yang dikirim ke Apps Script const payload = { tanggal: tanggalStruk, items: detectedItems, text: extractedText }; try { // PERBAIKAN: Menambahkan tanda kutip pada URL const response = await fetch('https://script.google.com/macros/s/AKfycbxIYfupSS1LyrzBa6_NViZhaxmUZ_SRbqNwyIgb9NtewPu5SnsAymiI8rX4YSGk5V1ZtA/exec', { method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8' }, body: JSON.stringify(payload) }); statusDiv.innerText = `Berhasil!
